    //JWT-account
    public static final String ACCOUNT = "username";
    //JWT-currentTimeMillis
    public final static String CURRENT_TIME_MILLIS = "currentTimeMillis";
    //æœ‰æ•ˆæœŸæ—¶é—´2å°æ—¶
    public static final long EXPIRE_TIME = 2 * 60 * 60 * 1000L;

    /**
     * ç”Ÿæˆç­¾åè¿”å›token

     * @param account

    public static String sign(String account, String currentTimeMillis) {
        // å¸å·åŠ JWTç§é’¥åŠ å¯†
        String secret = account + SECRET_KEY;
        // æ­¤å¤„è¿‡æœŸæ—¶é—´ï¼Œå•ä½ï¼šæ¯«ç§’ï¼Œåœ¨å½“å‰æ—¶é—´åˆ°åè¾¹çš„20åˆ†é’Ÿå†…éƒ½æ˜¯æœ‰æ•ˆçš„

        return JWT.create()
                .withClaim(ACCOUNT, account)
                .withClaim(CURRENT_TIME_MILLIS, currentTimeMillis)
                .withExpiresAt(date)
                //åˆ›å»ºä¸€ä¸ªæ–°çš„JWTï¼Œå¹¶ä½¿ç”¨ç»™å®šçš„ç®—æ³•è¿›è¡Œæ ‡è®°
                .sign(algorithm);
    }

    /**
     * æ ¡éªŒtokenæ˜¯å¦æ­£ç¡®
     *
     * @param token
     * @return
     */
    public static boolean verify(String token) {
        String secret = getClaim(token, ACCOUNT) + SECRET_KEY;
        Algorithm algorithm = Algorithm.HMAC256(secret);
        JWTVerifier verifier = JWT.require(algorithm)
                .build();
        verifier.verify(token);
        return true;
    }

    /**
     * è·å¾—Tokenä¸­çš„ä¿¡æ¯æ— éœ€secretè§£å¯†ä¹Ÿèƒ½è·å¾—

     * @param claim

    public static String getClaim(String token, String claim) {
        try {
            DecodedJWT jwt = JWT.decode(token);
            return jwt.getClaim(claim).asString();
        } catch (JWTDecodeException e) {
            return null;
        }
    }
